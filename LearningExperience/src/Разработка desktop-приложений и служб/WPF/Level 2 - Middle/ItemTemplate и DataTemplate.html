Все элементы управления списками имеют свойство ItemTemplate, которое позволяет задать свой шаблон отображения данных. В качестве значения оно принимает объект DataTemplate. В качестве самих данных, которые отображаются в DataTemplate, как правило, выступают объекты пользовательских классов, создаваемых самим разработчиком.


<ListBox x:Name="phonesList" SelectionChanged="phonesList_SelectionChanged">
<ListBox.ItemTemplate>
<DataTemplate>
<StackPanel Margin="5">
<Image Width="100" Height="75" Source="{Binding Path=ImagePath}" />
<TextBlock FontSize="16" Text="{Binding Path=Title}" HorizontalAlignment="Center" />
<TextBlock FontSize="16" Text="{Binding Path=Company}" HorizontalAlignment="Center" />
</StackPanel>
</DataTemplate>
</ListBox.ItemTemplate>
</ListBox>
